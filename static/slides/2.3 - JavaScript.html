JavaScript

  • Linked to HTML in two ways:
    • by reference:
      • src attribute on <script> element
    • by value:
      • inline JS inside <script> element
      • inline JS inside event attribute on element 

        <html>
          <body>
            

            <script src="./path/to/file.js"></script>

            <script>
              function onClick(msg) {
                alert('Pop-up message: ' + msg)
              }
            </script>
          </body>
        </html>

JavaScript example


    var heading = document.getElementsByTagName('h1')[0];
    var bulletPoints = document.getElementsByTagName('ul')[0].children;

    heading.addEventListener('mouseover', function (e) {
      modifyHeading(e.target)
    });
    modifyBulletPoints(bulletPoints);

    function modifyHeading(heading) {
      heading.textContent = 'This is a modified heading';
      heading.style = 'color: red';
    }

    function modifyBulletPoints(bulletPoints) {
      for (var i = 0; i < bulletPoints.length; i++) {
        bulletPoints[i].textContent += ' (modified)'
      }
    }
